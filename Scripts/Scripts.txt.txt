CREATE TABLE vendedor (
 
 COD_VENDEDOR int(11) NOT NULL AUTO_INCREMENT,

 NOM_VENDEDOR varchar(255) NOT NULL DEFAULT '' COMMENT 'Nome do Vendedor',

 CPF varchar(11) DEFAULT NULL,

 NUM_TELEFONE int(11) DEFAULT NULL,
  
 DAT_CADASTRO datetime NOT NULL,

 DAT_RETIRADA datetime DEFAULT NULL,

 PRIMARY KEY (COD_VENDEDOR)
) ENGINE=InnoDB AUTO_INCREMENT=52 DEFAULT CHARSET=utf8;
